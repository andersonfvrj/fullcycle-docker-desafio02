docker network desafio02

- Na pasta nginx
docker build -t andersonfvrj/desafio02-nginx .

- Na pasta node
docker build -t andersonfvrj/desafio02-node .

- Para rodar

docker run --rm -it --network desafio02 --name nodeserver -p 3000:3000 andersonfvrj/desafio02-node
docker run --rm -it --network desafio02 --name desafio02-nginx -p 8080:80 andersonfvrj/desafio02-nginx
